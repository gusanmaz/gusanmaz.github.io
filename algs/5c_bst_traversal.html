from collections import deque

class Node:
    def __init__(self, data):
        self.data = data
        self.left = None
        self.right = None

class BST:
    def __init__(self):
        self.root = None

    def insert(self, data):
        if not self.root:
            self.root = Node(data)
            return
        c = self.root
        while True:
            if data < c.data:
                if not c.left: c.left = Node(data); return
                c = c.left
            else:
                if not c.right: c.right = Node(data); return
                c = c.right

    # ===== TRAVERSAL METODLARI =====

    def inorder(self, node=None, result=None):
        """Sol â†’ KÃ¶k â†’ SaÄŸ (SÄ±ralÄ± Ã§Ä±ktÄ±)"""
        if result is None: result = []
        if node is None: node = self.root

        if node:
            self.inorder(node.left, result)   # Sol
            result.append(node.data)           # KÃ¶k
            self.inorder(node.right, result)  # SaÄŸ
        return result

    def preorder(self, node=None, result=None):
        """KÃ¶k â†’ Sol â†’ SaÄŸ"""
        if result is None: result = []
        if node is None: node = self.root

        if node:
            result.append(node.data)           # KÃ¶k
            self.preorder(node.left, result)  # Sol
            self.preorder(node.right, result) # SaÄŸ
        return result

    def postorder(self, node=None, result=None):
        """Sol â†’ SaÄŸ â†’ KÃ¶k"""
        if result is None: result = []
        if node is None: node = self.root

        if node:
            self.postorder(node.left, result)  # Sol
            self.postorder(node.right, result) # SaÄŸ
            result.append(node.data)            # KÃ¶k
        return result

    def level_order(self):
        """BFS ile seviye seviye dolaÅŸma"""
        if not self.root:
            return []

        result = []
        queue = deque([self.root])

        while queue:
            node = queue.popleft()  # FIFO
            result.append(node.data)

            if node.left:
                queue.append(node.left)
            if node.right:
                queue.append(node.right)

        return result
