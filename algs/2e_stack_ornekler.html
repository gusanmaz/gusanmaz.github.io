def parantez_kontrol(ifade):
    stack = []
    eslesme = {')': '(', ']': '[', '}': '{'}
    
    for char in ifade:
        if char in "({[":
            stack.append(char)
        elif char in ")}]":
            if not stack: return False
            son_acilan = stack.pop()
            if son_acilan != eslesme[char]:
                return False
                
    return len(stack) == 0
