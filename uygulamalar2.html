📝 Uygulama 1.2 — Satır Satır Okuma vs read() Tuzağı

🎯 Görev

Aşağıdaki kodu çalıştırın. read() ile tüm dosyayı RAM'e yüklemenin neden tehlikeli olabileceğini gözlemleyin. Sonra satır satır okuma ile karşılaştırın.

# Önce büyükçe bir dosya oluşturalım (1 milyon satır!)
import time

print("📝 1 milyon satırlık dosya oluşturuluyor...")
with open("buyuk_dosya.txt", "w") as f:
    for i in range(1_000_000):
        f.write(f"Satır {i}: Bu bir test satırıdır. " * 3 + "\n")

import os
boyut_mb = os.path.getsize("buyuk_dosya.txt") / (1024 * 1024)
print(f"📦 Dosya boyutu: {boyut_mb:.1f} MB")

# ❌ KÖTÜ YOL: Tamamını RAM'e yükle
start = time.time()
with open("buyuk_dosya.txt", "r") as f:
    tumu = f.read()  # Tüm dosya RAM'de!
sure_read = time.time() - start
print(f"\n⏱️ read() ile okuma: {sure_read:.3f} saniye")
print(f"📏 RAM'deki string uzunluğu: {len(tumu):,} karakter")

# ✅ İYİ YOL: Satır satır oku (lazy / generator)
start = time.time()
satir_sayisi = 0
with open("buyuk_dosya.txt", "r") as f:
    for satir in f:  # Her seferinde sadece 1 satır RAM'de
        satir_sayisi += 1
sure_satir = time.time() - start
print(f"\n⏱️ Satır satır okuma: {sure_satir:.3f} saniye")
print(f"📊 Toplam satır: {satir_sayisi:,}")

# Temizlik
os.remove("buyuk_dosya.txt")
print("\n🗑️ Dosya silindi.")
⚠️ Önemli Ders

f.read() tüm dosyayı bir anda belleğe yükler. 1 GB'lık bir dosyada bu, bilgisayarınızın RAM'ini doldurabilir! Her zaman satır satır okumayı tercih edin — özellikle dosya boyutunu bilmediğinizde. Bu, readlines() için de geçerlidir.

📝 Uygulama 1.3 — Dosyaya Ekleme (append) ve seek()

🎯 Görev

# Dosyamızı tekrar oluşturalım
with open("ogrenciler.txt", "w", encoding="utf-8") as f:
    f.write("Ahmet Yılmaz,85\n")
    f.write("Zeynep Kaya,92\n")

# "a" modu ile ekleme (append)
with open("ogrenciler.txt", "a", encoding="utf-8") as f:
    f.write("Yeni Öğrenci 1,76\n")
    f.write("Yeni Öğrenci 2,91\n")

# Dosyayı oku ve kontrol et
with open("ogrenciler.txt", "r", encoding="utf-8") as f:
    print("📄 Tüm içerik:")
    print(f.read())
    
    # seek(0) ile başa dön
    f.seek(0)
    ilk_50 = f.read(50)
    print(f"\n🔍 İlk 50 karakter:\n'{ilk_50}'")
    
    # tell() ile pozisyonu öğren
    print(f"\n📍 Şu anki pozisyon: {f.tell()}")

📝 Uygulama 2.2 — DictReader ile CSV Okuma

🎯 Görev

csv.DictReader kullanarak aynı dosyayı sözlük olarak okuyun. IMDB puanı 8.5'in üzerinde olan filmleri filtreleyin.

import csv

with open("filmler.csv", "r", encoding="utf-8") as f:
    reader = csv.DictReader(f)
    
    print("⭐ IMDB puanı 8.5+ olan filmler:\n")
    for film in reader:
        if float(film["IMDB"]) >= 8.5:
            print(f"  🎬 {film['Başlık']} ({film['Yıl']}) — ⭐ {film['IMDB']}")
            print(f"     Yönetmen: {film['Yönetmen']} | Tür: {film['Tür']}\n")

🔥 Uygulama 3.3 — Gerçek JSON API: Dünya Ülkeleri

🎯 Görev

Gerçek bir API'den JSON verisi çekin ve analiz edin. REST Countries API ücretsiz ve açık bir API'dir.

import json
from urllib.request import urlopen

# REST Countries API'den tüm ülkeleri çek
url = "https://restcountries.com/v3.1/all?fields=name,capital,population,region,area,languages"
print("🌐 API'den veri çekiliyor...")
response = urlopen(url)
ulkeler = json.loads(response.read().decode("utf-8"))

print(f"✅ {len(ulkeler)} ülke yüklendi!\n")

# İlk ülkenin yapısını inceleyelim
print("🔍 Örnek veri yapısı (ilk ülke):")
print(json.dumps(ulkeler[0], indent=2, ensure_ascii=False)[:500])

# Nüfusa göre sırala
ulkeler.sort(key=lambda x: x.get("population", 0), reverse=True)

print("\n🌍 Dünya'nın en kalabalık 15 ülkesi:")
for i, u in enumerate(ulkeler[:15], 1):
    ad = u["name"]["common"]
    nufus = u.get("population", 0)
    baskent = u.get("capital", ["?"])[0] if u.get("capital") else "?"
    bolge = u.get("region", "?")
    print(f"  {i:2}. {ad:30s} | 🏙️ {baskent:20s} | 👥 {nufus:>15,} | 🌏 {bolge}")

# Bölgelere göre ülke sayısı
print("\n📊 Bölgelere göre ülke sayısı:")
bolgeler = {}
for u in ulkeler:
    b = u.get("region", "Bilinmiyor")
    bolgeler[b] = bolgeler.get(b, 0) + 1

for bolge, sayi in sorted(bolgeler.items(), key=lambda x: x[1], reverse=True):
    print(f"  🌏 {bolge:20s}: {sayi} ülke")

# Türkiye'yi bulalım
turkiye = [u for u in ulkeler if "Turkey" in u["name"].get("common", "") or "Türkiye" in u["name"].get("common", "")]
if turkiye:
    t = turkiye[0]
    print(f"\n🇹🇷 Türkiye Bilgileri:")
    print(json.dumps(t, indent=2, ensure_ascii=False))

🔥 Uygulama 3.4 — Büyük JSON: Nobel Ödülleri

📥 Veri Seti

Nobel Ödülü API'si açık bir JSON API'dir. Oldukça büyük ve zengin bir veri seti:

📎 Nobel Prize API — prize.json

import json
from urllib.request import urlopen
from collections import Counter

url = "https://api.nobelprize.org/v1/prize.json"
print("🏅 Nobel ödülleri verisi çekiliyor...")
response = urlopen(url)
data = json.loads(response.read().decode("utf-8"))
oduller = data["prizes"]

print(f"✅ Toplam {len(oduller)} ödül kaydı yüklendi!")

# Yapıyı incele
print("\n🔍 Örnek kayıt:")
print(json.dumps(oduller[0], indent=2, ensure_ascii=False))

# Kategorilere göre ödül sayısı
kategoriler = Counter(o["category"] for o in oduller)
print("\n📊 Kategorilere göre ödül sayısı:")
for kat, sayi in kategoriler.most_common():
    print(f"  🏅 {kat:20s}: {sayi}")

# Çok ödüllü kişiler (birden fazla Nobel alanlar)
kisiler = {}
for odul in oduller:
    for kazanan in odul.get("laureates", []):
        ad = f"{kazanan.get('firstname', '')} {kazanan.get('surname', '')}".strip()
        if ad:
            kisiler[ad] = kisiler.get(ad, 0) + 1

cok_odullu = {k: v for k, v in kisiler.items() if v > 1}
print(f"\n🌟 Birden fazla Nobel ödülü alan kişiler ({len(cok_odullu)} kişi):")
for ad, sayi in sorted(cok_odullu.items(), key=lambda x: x[1], reverse=True):
    print(f"  ⭐ {ad}: {sayi} ödül")

# Son 5 yılın fizik ödüllerini listele
print("\n⚛️ Son 5 yılın Fizik Nobel ödülleri:")
fizik = [o for o in oduller if o["category"] == "physics"]
fizik.sort(key=lambda x: int(x["year"]), reverse=True)
for o in fizik[:5]:
    kazananlar = ", ".join(
        f"{k.get('firstname', '')} {k.get('surname', '')}" 
        for k in o.get("laureates", [])
    )
    motivasyon = o.get("laureates", [{}])[0].get("motivation", "")
    print(f"  {o['year']}: {kazananlar}")
    if motivasyon:
        print(f"         📜 {motivasyon[:100]}...")

🔥 Uygulama 3.5 — JSON Lines (JSONL) Formatı

🎯 Görev

JSON Lines formatı büyük veri dünyasında çok kullanılır: her satır bağımsız bir JSON objesidir. Bu format büyük dosyaları satır satır okumaya olanak tanır (tüm dosyayı RAM'e yüklemek zorunda kalmazsınız).

import json

# JSONL dosyası oluştur: her satır bir JSON objesi
log_kayitlari = [
    {"timestamp": "2025-03-01T10:30:00", "user": "ahmet", "action": "login", "ip": "192.168.1.10"},
    {"timestamp": "2025-03-01T10:31:15", "user": "ahmet", "action": "view_page", "page": "/dashboard"},
    {"timestamp": "2025-03-01T10:32:00", "user": "zeynep", "action": "login", "ip": "192.168.1.42"},
    {"timestamp": "2025-03-01T10:33:22", "user": "ahmet", "action": "download", "file": "rapor.pdf"},
    {"timestamp": "2025-03-01T10:35:00", "user": "zeynep", "action": "upload", "file": "veri.csv", "size_mb": 15.2},
    {"timestamp": "2025-03-01T10:36:10", "user": "mehmet", "action": "login", "ip": "10.0.0.5"},
    {"timestamp": "2025-03-01T10:40:00", "user": "ahmet", "action": "logout"},
    {"timestamp": "2025-03-01T10:45:30", "user": "zeynep", "action": "view_page", "page": "/analytics"},
    {"timestamp": "2025-03-01T10:50:00", "user": "mehmet", "action": "error", "error": "PermissionDenied", "page": "/admin"},
    {"timestamp": "2025-03-01T10:55:00", "user": "zeynep", "action": "logout"},
]

# JSONL dosyasına yaz
with open("server_log.jsonl", "w", encoding="utf-8") as f:
    for kayit in log_kayitlari:
        f.write(json.dumps(kayit, ensure_ascii=False) + "\n")

print("✅ server_log.jsonl oluşturuldu!\n")

# JSONL dosyasını satır satır oku (bellek dostu!)
print("📋 Log kayıtları:")
with open("server_log.jsonl", "r", encoding="utf-8") as f:
    for satir in f:
        kayit = json.loads(satir.strip())
        emoji = {"login": "🔑", "logout": "🚪", "view_page": "👁️", 
                 "download": "⬇️", "upload": "⬆️", "error": "❌"}.get(kayit["action"], "📌")
        print(f"  {emoji} [{kayit['timestamp'][11:]}] {kayit['user']:10s} → {kayit['action']}")

# Kullanıcı başına aksiyon sayısı
print("\n📊 Kullanıcı bazlı istatistikler:")
with open("server_log.jsonl", "r") as f:
    kullanici_aksiyonlar = {}
    for satir in f:
        k = json.loads(satir)
        user = k["user"]
        kullanici_aksiyonlar.setdefault(user, []).append(k["action"])

for user, aksiyonlar in kullanici_aksiyonlar.items():
    print(f"  👤 {user}: {len(aksiyonlar)} aksiyon — {', '.join(aksiyonlar)}")

import pandas as pd

    "Matematik": [85, 92, 78, 95, 88, 72, 90, 84],

print(df.sort_values("Ortalama", ascending=False))

        "buyukluk": props.get("mag", 0),
        "zaman": pd.to_datetime(props.get("time", 0), unit="ms"),
        "derinlik_km": coords[2] if len(coords) > 2 else None,
        "boylam": coords[0],
        "enlem": coords[1],
        "tsunami": props.get("tsunami", 0),
    })

df = pd.DataFrame(depremler)

for _, d in en_buyukler.iterrows():

    print(f"  🔴 {d['buyukluk']:.1f} — {d['yer']} — {d['zaman'].strftime('%Y-%m-%d %H:%M')} {tsunami}")

# Tsunami uyarısı olan depremler

🏆 Uygulama 5.1 — CSV ↔ JSON Dönüştürücü

🎯 Görev

İki fonksiyon yazın: biri CSV'yi JSON'a, diğeri JSON'u CSV'ye dönüştürsün. Her iki yönde de çalışmalı.

import csv
import json

def csv_to_json(csv_dosya, json_dosya):
    """CSV dosyasını JSON'a dönüştürür."""
    with open(csv_dosya, "r", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        satirlar = list(reader)
    
    with open(json_dosya, "w", encoding="utf-8") as f:
        json.dump(satirlar, f, ensure_ascii=False, indent=2)
    
    print(f"✅ {csv_dosya} → {json_dosya} ({len(satirlar)} kayıt)")
    return satirlar

def json_to_csv(json_dosya, csv_dosya):
    """JSON dosyasını CSV'ye dönüştürür."""
    with open(json_dosya, "r", encoding="utf-8") as f:
        veri = json.load(f)
    
    if not veri:
        print("❌ Boş veri!")
        return
    
    with open(csv_dosya, "w", newline="", encoding="utf-8") as f:
        writer = csv.DictWriter(f, fieldnames=veri[0].keys())
        writer.writeheader()
        writer.writerows(veri)
    
    print(f"✅ {json_dosya} → {csv_dosya} ({len(veri)} kayıt)")
    return veri

# Test: Önce CSV oluştur
with open("test_veri.csv", "w", newline="", encoding="utf-8") as f:
    writer = csv.writer(f)
    writer.writerow(["sehir", "ulke", "nufus", "alan_km2"])

🏆 Uygulama 5.3 — Mega Challenge: Veri Pipeline'ı

🔥 Zorlayıcı Görev

Bu görev tüm öğrendiklerinizi birleştirir. Baştan sona bir veri pipeline'ı oluşturun:

  1. Veri Toplama: REST Countries API'den tüm ülkeleri JSON olarak çekin
  2. Veri Temizleme: Sadece şu sütunları alın: isim, başkent, nüfus, bölge, alan (km²), diller
  3. Dönüştürme: Nüfus yoğunluğunu hesaplayın (nüfus / alan)
  4. Filtreleme: Nüfusu 1 milyonun altında olan ülkeleri çıkarın
  5. Analiz: Bölge bazlı ortalama nüfus yoğunluğunu hesaplayın
  6. Kaydetme: Sonuçları 3 farklı formatta kaydedin: CSV, JSON ve JSONL
  7. Doğrulama: Kaydedilen 3 dosyayı tekrar okuyup satır sayılarının eşit olduğunu kontrol edin
import json
import csv
import pandas as pd
from urllib.request import urlopen

# ═══ 1. VERİ TOPLAMA ═══
print("🌐 1. Veri toplama...")
url = "https://restcountries.com/v3.1/all"
response = urlopen(url)
raw = json.loads(response.read().decode("utf-8"))
print(f"   ✅ {len(raw)} ülke çekildi")

# ═══ 2. VERİ TEMİZLEME ═══
print("🧹 2. Veri temizleme...")
temiz = []
for u in raw:
    try:
        temiz.append({
            "ulke": u["name"]["common"],
            "baskent": u.get("capital", ["?"])[0] if u.get("capital") else "?",
            "nufus": u.get("population", 0),
            "bolge": u.get("region", "Bilinmiyor"),
            "alan_km2": u.get("area", 0),
            "diller": ", ".join(u.get("languages", {}).values()) if u.get("languages") else "?"
        })
    except:
        pass

df = pd.DataFrame(temiz)
print(f"   ✅ {len(df)} kayıt temizlendi")

# ═══ 3. DÖNÜŞTÜRME ═══
print("🔄 3. Nüfus yoğunluğu hesaplanıyor...")
df["yogunluk"] = (df["nufus"] / df["alan_km2"]).round(1)
df["yogunluk"] = df["yogunluk"].fillna(0)

# ═══ 4. FİLTRELEME ═══
print("🔍 4. Nüfusu 1M+ olan ülkeler filtreleniyor...")
df = df[df["nufus"] >= 1_000_000].copy()
df = df.sort_values("nufus", ascending=False).reset_index(drop=True)
print(f"   ✅ {len(df)} ülke kaldı")

# ═══ 5. ANALİZ ═══
print("\n📊 5. Bölge bazlı analiz:")
analiz = df.groupby("bolge").agg(
    ulke_sayisi=("ulke", "count"),
    toplam_nufus=("nufus", "sum"),
    ort_yogunluk=("yogunluk", "mean")
).round(1)
print(analiz)

print("\n🏙️ En yoğun 10 ülke:")
for _, r in df.nlargest(10, "yogunluk").iterrows():
    print(f"  {r['ulke']:25s} — {r['yogunluk']:>10,.1f} kişi/km²")

# ═══ 6. KAYDETME ═══
print("\n💾 6. Dosyalara kaydediliyor...")

# CSV
df.to_csv("ulkeler_analiz.csv", index=False, encoding="utf-8")
print("   ✅ ulkeler_analiz.csv")

# JSON
df.to_json("ulkeler_analiz.json", orient="records", force_ascii=False, indent=2)
print("   ✅ ulkeler_analiz.json")

# JSONL
with open("ulkeler_analiz.jsonl", "w", encoding="utf-8") as f:
    for _, row in df.iterrows():
        f.write(json.dumps(row.to_dict(), ensure_ascii=False) + "\n")
print("   ✅ ulkeler_analiz.jsonl")

# ═══ 7. DOĞRULAMA ═══
print("\n✔️ 7. Doğrulama...")
df_csv = pd.read_csv("ulkeler_analiz.csv")
df_json = pd.read_json("ulkeler_analiz.json")
with open("ulkeler_analiz.jsonl", "r") as f:
    jsonl_satirlar = [json.loads(line) for line in f]

print(f"   CSV:   {len(df_csv)} satır")
print(f"   JSON:  {len(df_json)} satır")
print(f"   JSONL: {len(jsonl_satirlar)} satır")

import os
for dosya in ["ulkeler_analiz.csv", "ulkeler_analiz.json", "ulkeler_analiz.jsonl"]:
    boyut = os.path.getsize(dosya)
    print(f"   📦 {dosya}: {boyut:,} byte")

if len(df_csv) == len(df_json) == len(jsonl_satirlar):
    print("\n🎉 Tüm dosyalar tutarlı! Pipeline başarılı!")

🏋️ Pandas ile Büyük Dosya Okuma — Chunk chunk okuma örneği Aşağıdaki kodu kullanın ↓ 
# Pandas ile büyük dosyaları chunk chunk okuma
import pandas as pd

# chunksize parametresi ile büyük dosyayı parça parça oku
# Her seferinde sadece 10.000 satır RAM'e yüklenir
chunk_sayisi = 0
toplam_satir = 0

for chunk in pd.read_csv("buyuk_dosya.csv", chunksize=10_000):
    chunk_sayisi += 1
    toplam_satir += len(chunk)
    
    # Her chunk bir DataFrame — istediğiniz işlemi yapın
    # Örnek: sadece belirli sütunlardaki ortalamayı hesapla
    # ortalama = chunk["fiyat"].mean()
    
    if chunk_sayisi <= 3:
        print(f"  Chunk {chunk_sayisi}: {len(chunk)} satır — Bellek dostu! ✅")

print(f"\n📊 Toplam: {chunk_sayisi} chunk, {toplam_satir:,} satır")
print("🧠 RAM'de hiçbir zaman 10.000'den fazla satır tutulmadı!")

📁 Dosya Okuma/Yazma

# Yazma
with open("dosya.txt", "w", encoding="utf-8") as f:
    f.write("merhaba\n")

# Okuma (tamamı — dikkat: büyük dosyalarda RAM sorunu!)
with open("dosya.txt", "r", encoding="utf-8") as f:
    icerik = f.read()

# Satır satır okuma (bellek dostu ✅)
with open("dosya.txt", "r") as f:
    for satir in f:
        print(satir.strip())

# Ekleme (append)
with open("dosya.txt", "a") as f:
    f.write("yeni satir\n")

# Dosya modları: "r" oku, "w" yaz (sil+yaz), "a" ekle, "rb"/"wb" binary

📊 CSV İşlemleri

import csv

# CSV yazma
with open("veri.csv", "w", newline="", encoding="utf-8") as f:
    writer = csv.writer(f)
    writer.writerow(["ad", "yas"])   # başlık
    writer.writerow(["Ahmet", 25])   # veri

# CSV okuma
with open("veri.csv", "r", encoding="utf-8") as f:
    reader = csv.reader(f)
    for satir in reader:
        print(satir)

# DictReader (sözlük olarak oku)
with open("veri.csv", "r", encoding="utf-8") as f:
    reader = csv.DictReader(f)
    for kayit in reader:
        print(kayit["ad"], kayit["yas"])

📦 JSON İşlemleri

import json

# Python → JSON dosya
veri = {"ad": "Ahmet", "yas": 25, "dersler": ["BDT", "YBS"]}
with open("veri.json", "w", encoding="utf-8") as f:
    json.dump(veri, f, ensure_ascii=False, indent=2)

# JSON dosya → Python
with open("veri.json", "r", encoding="utf-8") as f:
    veri = json.load(f)

# String ↔ JSON
json_str = json.dumps(veri, ensure_ascii=False)  # dict → string
veri = json.loads(json_str)                        # string → dict

# API'den JSON okuma
from urllib.request import urlopen

import pandas as pd

# Okuma
df = pd.read_csv("veri.csv")
df = pd.read_json("veri.json")
df = pd.read_csv("https://url.com/veri.csv")  # URL'den

# Keşif
df.head()           # İlk 5 satır
df.tail()           # Son 5 satır
df.shape            # (satır, sütun)
df.columns          # Sütun isimleri
df.dtypes           # Veri tipleri
df.info()           # Genel bilgi
df.describe()       # İstatistikler
df.isnull().sum()   # Eksik veri sayısı

# Filtreleme
df[df["yas"] > 25]                              # Tek koşul
df[(df["yas"] > 25) & (df["sehir"] == "İst")]   # Çoklu koşul

# Sıralama
df.sort_values("yas", ascending=False)

# Gruplama
df.groupby("sehir")["maas"].mean()

# Kaydetme
df.to_csv("sonuc.csv", index=False)
df.to_json("sonuc.json", orient="records", force_ascii=False)
