    def __init__(self):

    def get(self, key, hesaplama_func):
        """
        Cache'te varsa getir, yoksa hesapla ve sakla

        if key in self.cache:

            self.hit_count += 1

            self.miss_count += 1

            value = hesaplama_func(key)

        print(f"   Hit: {self.hit_count} (%{hit_rate:.1f})")
